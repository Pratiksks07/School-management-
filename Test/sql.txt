Create database smdb;

use smdb;

Create table login
(
userid varchar(15) Primary key,
passwd varchar(15),
type char(2),
name varchar(30)
);

create table students
(
userid varchar(15) primary key,
name varchar(30),
roll int(3),
class int(2),
sec char(1),
fname varchar(30),
mname varchar(30),
address varchar(60),
phone bigint,
admno int(10),
passwd varchar(15),
totaldays int,
present int,
absent int,
percent int
);

create table message
(
userid varchar(15) primay key,
name varchar(30),
type char(2),
message longtext,
assignment longblob
);
